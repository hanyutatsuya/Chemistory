# vi /etc/php-fpm.d/www.conf

↓
clear_env = no

env[LANG] = ja_JP.SJIS

env[DB2_HOME] = /home/db2inst1/sqllib
env[DB2LIB] = /home/db2inst1/sqllib/lib
env[IBM_DB_LIB] = /home/db2inst1/sqllib/lib
env[IBM_DB_HOME] = /home/db2inst1/sqllib
env[IBM_DB_DIR] = /home/db2inst1/sqllib
env[IBM_DB_INCLUDE] = /home/db2inst1/sqllib/include
env[PATH] = /home/db2inst1/sqllib/bin:/usr/local/bin:/usr/bin:/usr/local/sbin:/usr/sbin
env[LD_LIBRARY_PATH] = /home/db2inst1/sqllib/lib64:/home/db2inst1/sqllib/lib64/gskit:/home/db2inst1/sqllib/lib32
